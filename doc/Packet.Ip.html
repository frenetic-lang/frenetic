Module Packet.Ip

module Ip: sig .. end
IPv4 frame of a packet.
type tp =
| Tcp of Packet.Tcp.t
| Icmp of Packet.Icmp.t
| Unparsable of (Packet.nwProto * Packet.bytes)
tp is the type of protocol, which indicates which protocol is encapsulated in the payload of the IPv4 frame. At present, only TCP and ICMP are explicitly supported; otherwise, the raw bytes and IPv4 protocol number are provided.
module Flags: sig .. end
type t = {
   tos :Packet.nwTos; (*Type of service.*)
   ident :Packet.int16; (*Identification.*)
   flags :Flags.t; (*IPv4 header flags.*)
   frag :Packet.int16; (*Fragment offset.*)
   ttl :Packet.int8; (*Time to live.*)
   chksum :Packet.int16; (*Header checksum.*)
   src :Packet.nwAddr; (*IP source address.*)
   dst :Packet.nwAddr; (*IP destination address.*)
   tp :tp; (*Packet payload.*)
}
